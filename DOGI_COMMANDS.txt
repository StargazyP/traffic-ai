5) 선택: 초기 docker-compose.yml 생성 템플릿
==================================================
cat > docker-compose.yml <<'EOF'
services:
  app:
    image: python:3.11-slim
    working_dir: /workspace
    volumes:
      - ./:/workspace
    command: bash -lc "python -V && sleep infinity"
EOF

docker compose up -d
docker compose exec app sh
